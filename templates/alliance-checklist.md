## 自动化脚本

```bash
#!/bin/bash
# alliance-check.sh - 联盟检查脚本

POD_PATH="/path/to/cryopod.enc"
EXPECTED_HASH="expected_sha256_hash"

# 计算哈希
ACTUAL_HASH=$(sha256sum "$POD_PATH" | cut -d' ' -f1)

# 对比
if [ "$ACTUAL_HASH" == "$EXPECTED_HASH" ]; then
    echo "✅ 冷冻包完整性验证通过"
    exit 0
else
    echo "❌ 冷冻包完整性验证失败"
    echo "   预期：$EXPECTED_HASH"
    echo "   实际：$ACTUAL_HASH"
    exit 1
fi
```
