```python
def scan_all_domains():
    """扫描 knowledge_base 下所有领域目录"""
    domains = []
    for item in KNOWLEDGE_BASE_DIR.iterdir():
        if item.is_dir() and re.match(r'^\d{2}-', item.name):
            domains.append(item.name)
        elif item.name == '23_articles_series':
            domains.append(item.name)
    return sorted(domains)
```

#### 知识点计数修复
```python
# 旧：简单计数
count = len(files)

# 新：解析"**数量**:" 元数据
def count_points_in_file(filepath):
    """解析文件中的知识点数量"""
    with open(filepath, 'r', encoding='utf-8') as f:
        content = f.read(2000)  # 只读前 2000 字符
        match = re.search(r'\*\*数量\*\*:\s*(\d+)', content)
        return int(match.group(1)) if match else 1
```

### 3. 更新 auto_sync.py

#### 扫描范围扩展
```python
# 旧：固定领域列表
DOMAINS = ['01-ai-agent', '02-openclaw', ..., '12-tools']

# 新：动态扫描
def discover_domains():
    """自动发现所有领域目录"""
    domains = []
    for item in KNOWLEDGE_BASE_DIR.iterdir():
        if item.is_dir():
            # 匹配 XX-xxx 格式或 23_articles_series
            if re.match(r'^\d{2}-', item.name) or 'articles' in item.name:
                domains.append(item.name)
    return domains
```
